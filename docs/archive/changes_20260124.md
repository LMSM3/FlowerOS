check_gcc() {
    if ! command -v gcc >/dev/null 2>&1; then
        err "gcc not found!"
        echo ""
        echo "Install with:"
        echo "  Ubuntu/Debian: sudo apt install build-essential"
        echo "  Fedora/RHEL:   sudo dnf install gcc"
        echo "  Arch:          sudo pacman -S gcc"
        echo ""
        return 1
    fi
    ok "Found gcc: $(gcc --version | head -n1)"
    return 0
}
